```python
def _calc_rates(*, sent_today: int, replies_today: int,
                avg_sent_7d: float, avg_replies_7d: float,
                bounced: int, bounce_sent: int) -> tuple:
    """Calculate reply rates and bounce rate. Returns (reply_rate_today, reply_rate_7d, bounce_rate)."""
    reply_rate_today = (replies_today / sent_today * 100) if sent_today > 0 else 0.0
    reply_rate_7d = (avg_replies_7d / avg_sent_7d * 100) if avg_sent_7d > 0 else 0.0
    bounce_rate = (bounced / bounce_sent * 100) if bounce_sent > 0 else 0.0
    return (reply_rate_today, reply_rate_7d, bounce_rate)


def _build_client_data(*, platform: str,
                       active_campaigns: int, total_campaigns: int,
                       sent_today: int, first_touch_today: int, followup_today: int,
                       replies_today: int, opps_today: int,
                       reply_rate_today: float, reply_rate_7d: float, bounce_rate: float,
                       not_contacted: int, in_progress,
                       avg_sent_7d: float, avg_replies_7d: float, avg_opps_7d: float,
                       campaigns: list) -> dict:
    """Assemble a ClientData dict from fetcher-provided values. Adds trend indicators via _trend()."""
    return {
        "platform": platform,
        "active_campaigns": active_campaigns,
        "total_campaigns": total_campaigns,
        "sent_today": sent_today,
        "first_touch_today": first_touch_today,
        "followup_today": followup_today,
        "replies_today": replies_today,
        "opps_today": opps_today,
        "reply_rate_today": round(reply_rate_today, 2),
        "reply_rate_7d": round(reply_rate_7d, 2),
        "bounce_rate": round(bounce_rate, 2),
        "not_contacted": not_contacted,
        "in_progress": in_progress,
        "avg_sent_7d": round(avg_sent_7d, 1),
        "avg_replies_7d": round(avg_replies_7d, 1),
        "avg_opps_7d": round(avg_opps_7d, 1),
        "opp_trend": _trend(opps_today, avg_opps_7d),
        "reply_trend": _trend(reply_rate_today, reply_rate_7d),
        "sent_trend": _trend(sent_today, avg_sent_7d),
        "campaigns": campaigns,
    }
```
