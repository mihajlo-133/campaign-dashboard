```python
# ---------------------------------------------------------------------------
# Data contracts — TypedDict definitions for fetcher return shapes
# ---------------------------------------------------------------------------

from typing import TypedDict, NotRequired

class CampaignData(TypedDict):
    name: str
    id: str
    status: str
    sent_today: int
    first_touch: int
    followups: int
    replies_today: int
    opps_today: int
    reply_rate: float
    not_contacted: int
    in_progress: int
    total_sent: int
    total_bounced: int
    # Optional (Instantly-only)
    total_leads: NotRequired[int]
    total_completed: NotRequired[int]

class DailyData(TypedDict):
    date: str
    sent: int
    opps: int
    replies: int

class ClientData(TypedDict):
    platform: str
    active_campaigns: int
    total_campaigns: int
    sent_today: int
    first_touch_today: int
    followup_today: int
    replies_today: int
    opps_today: int
    reply_rate_today: float
    reply_rate_7d: float
    bounce_rate: float
    not_contacted: int
    in_progress: int       # int or None for EmailBison
    avg_sent_7d: float
    avg_replies_7d: float
    avg_opps_7d: float
    opp_trend: str
    reply_trend: str
    sent_trend: str
    campaigns: list
    # Optional / platform-specific
    daily: NotRequired[list]
    _nc_backfill: NotRequired[list]
    _nc_api_key: NotRequired[str]
```

Note: Add `from typing import TypedDict, NotRequired` to the imports at the top of server.py (after the existing imports, before the Constants section). Python 3.11+ is required for `NotRequired` -- this is fine since the project targets 3.11+.
